def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
    m1 = defaultdict(int)
    m2 = defaultdict(int)
    res = []
    for i in nums1:
        m1[i] +=1
    for j in nums2:
        m2[j] +=1
    for i in m1:
        if i in m2:
            # If we only need the unique values in the intersection
            res += [i] 
            # If we nedd all the intersection no matter what
            res += [i]*min(m1[i],m2[i])
    return res
