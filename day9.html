class MyQueue:

            self.stack1 = []
            self.stack2 = []

            self.stack1.append(x)

            peek = self.peek()
            return self.stack2.pop()
            
    
        def peek(self) -> int:
            if self.stack2: return self.stack2[-1]
            if not self.stack1: return -1
            while self.stack1:
                self.stack2.append(self.stack1.pop())
            return self.stack2[-1]
    
        def empty(self) -> bool:
            return not self.stack1 and not self.stack2

Similar to problem4, when you have to consider the orders of incoming parentheses, use stacks.


class Solution:
        def isValid(self, s: str) -> bool:
            m  = {'(':')','{':'}','[':']'}
            stack = []
            for i in s:
                if i in m.keys():
                    stack.append(i)
                else:
                    if not stack:
                        return False
                    elif i== m[stack.pop()]:
                        continue
                    else:
                        return False
            return stack ==[]

The trick is considering when you vanish repetitive letters there can be new repetitives generated! Use stack to keep the result then!


def removeDuplicates(self, s: str) -> str:
    res = []
    for a in s:
        if res:
            if a ==res[-1]:
                res.pop()
                continue
        res.append(a)
    return ''.join(res)
